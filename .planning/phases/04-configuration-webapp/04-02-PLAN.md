       ```javascript
       const POLL_INTERVAL = 2000;
       let previousRegValues = {};

       // -- Status polling --
       async function pollStatus() {
         try {
           const res = await fetch('/api/status');
           const data = await res.json();
           const seDot = document.getElementById('se-dot');
           const seLabel = document.getElementById('se-label');
           seDot.className = 'status-dot';
           if (data.reconfiguring) {
             seDot.classList.add('yellow');
             seLabel.textContent = 'SolarEdge: Reconnecting...';
           } else if (data.solaredge === 'connected') {
             seDot.classList.add('green');
             seLabel.textContent = 'SolarEdge: Connected';
           } else if (data.solaredge === 'night_mode') {
             seDot.classList.add('yellow');
             seLabel.textContent = 'SolarEdge: Night Mode';
           } else {
             seDot.classList.add('red');
             seLabel.textContent = 'SolarEdge: ' + (data.solaredge || 'Disconnected');
           }
           const vosDot = document.getElementById('vos-dot');
           const vosLabel = document.getElementById('vos-label');
           vosDot.className = 'status-dot';
           if (data.venus_os === 'active') {
             vosDot.classList.add('green');
             vosLabel.textContent = 'Venus OS: Active';
           } else {
             vosDot.classList.add('yellow');
             vosLabel.textContent = 'Venus OS: ' + (data.venus_os || 'Unknown');
           }
         } catch (e) { console.error('Status poll failed:', e); }
       }

       // -- Health polling --
       async function pollHealth() {
         try {
           const res = await fetch('/api/health');
           const data = await res.json();
           const hrs = Math.floor(data.uptime_seconds / 3600);
           const mins = Math.floor((data.uptime_seconds % 3600) / 60);
           document.getElementById('uptime').textContent = hrs + 'h ' + mins + 'm';
           document.getElementById('poll-rate').textContent = data.poll_success_rate.toFixed(1) + '%';
           if (data.last_poll_age !== null) {
             document.getElementById('last-poll').textContent = data.last_poll_age.toFixed(0) + 's ago';
           } else {
             document.getElementById('last-poll').textContent = 'No data';
           }
           document.getElementById('cache-status').textContent = data.cache_stale ? 'STALE' : 'Fresh';
           document.getElementById('cache-status').style.color = data.cache_stale ? 'var(--red)' : 'var(--green)';
         } catch (e) { console.error('Health poll failed:', e); }
       }

       // -- Config loading --
       async function loadConfig() {
         try {
           const res = await fetch('/api/config');
           const data = await res.json();
           document.getElementById('se-host').value = data.host;
           document.getElementById('se-port').value = data.port;
           document.getElementById('se-unit').value = data.unit_id;
         } catch (e) { console.error('Config load failed:', e); }
       }

       // -- Test connection --
       document.getElementById('btn-test').addEventListener('click', async () => {

         } catch (e) {
           msg.className = 'message error';
           msg.textContent = 'Test request failed: ' + e.message;
         }
       });

       // -- Save config --
       document.getElementById('config-form').addEventListener('submit', async (e) => {
         e.preventDefault();
         const msg = document.getElementById('config-message');
         msg.className = 'message';

         msg.textContent = 'Saving...';

             method: 'POST',
             headers: {'Content-Type': 'application/json'},
             body: JSON.stringify({
               host: document.getElementById('se-host').value,
               port: parseInt(document.getElementById('se-port').value),
               unit_id: parseInt(document.getElementById('se-unit').value)
             })
           });
           const data = await res.json();
           msg.className = 'message ' + (data.success ? 'success' : 'error');
           msg.textContent = data.success ? 'Saved and applied!' : 'Save failed: ' + data.error;
         } catch (e) {
           msg.className = 'message error';
           msg.textContent = 'Save request failed: ' + e.message;
         }
       });

       // -- Register viewer --
       async function pollRegisters() {
         try {
           const res = await fetch('/api/registers');
           const models = await res.json();
           const container = document.getElementById('register-models');
           // Build HTML on first load, update values on subsequent polls
           if (container.children.length === 0) {
             buildRegisterViewer(container, models);
           } else {
             updateRegisterValues(models);
           }
         } catch (e) { console.error('Register poll failed:', e); }
       }

       function buildRegisterViewer(container, models) {
         models.forEach((model, idx) => {
           const group = document.createElement('div');
           group.className = 'model-group';
           const header = document.createElement('div');
           header.className = 'model-header';
           header.innerHTML = '<span>' + model.name + '</span><span>&#9660;</span>';
           header.addEventListener('click', () => {
             const fields = group.querySelector('.model-fields');
             fields.classList.toggle('collapsed');
             header.querySelector('span:last-child').textContent = fields.classList.contains('collapsed') ? '\u25B6' : '\u25BC';
           });
           group.appendChild(header);

           const fields = document.createElement('div');
           fields.className = 'model-fields';
           model.fields.forEach(field => {
             const row = document.createElement('div');
             row.className = 'reg-row';
             row.id = 'reg-' + field.addr;
             row.innerHTML = '<span class="reg-addr">' + field.addr + '</span>' +
               '<span class="reg-name">' + field.name + '</span>' +
               '<span class="reg-value" id="val-' + field.addr + '">' + formatValue(field.value) + '</span>';
             fields.appendChild(row);
             previousRegValues[field.addr] = field.value;
           });
           group.appendChild(fields);
           container.appendChild(group);
         });
       }

       function updateRegisterValues(models) {
         models.forEach(model => {
           model.fields.forEach(field => {
             const valEl = document.getElementById('val-' + field.addr);
             if (valEl) {
               const newVal = formatValue(field.value);
               if (valEl.textContent !== newVal) {
                 valEl.textContent = newVal;
                 const row = document.getElementById('reg-' + field.addr);
                 if (row) {
                   row.classList.remove('changed');
                   void row.offsetWidth; // force reflow for re-animation
                   row.classList.add('changed');
                 }
               }
               previousRegValues[field.addr] = field.value;
             }
           });
         });
       }

       function formatValue(val) {
         if (typeof val === 'string') return val;
         if (val === null || val === undefined) return '--';
         return val.toString();
       }

       // -- Uptime formatting helper --
       function formatUptime(seconds) {
         const h = Math.floor(seconds / 3600);
         const m = Math.floor((seconds % 3600) / 60);
         return h + 'h ' + m + 'm';
       }

       // -- Initial load + polling --
       loadConfig();
       pollStatus();
       pollHealth();
       pollRegisters();
       setInterval(() => { pollStatus(); pollHealth(); pollRegisters(); }, POLL_INTERVAL);
       ```
