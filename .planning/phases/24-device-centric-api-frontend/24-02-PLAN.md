```json
    {"id": "virtual", "name": "Fronius PV Inverter Proxy", "type": "virtual"}
  ]
}
```

GET /api/devices/{id}/snapshot -> JSON:
```json
{
  "device_id": "abc123",
  "device_type": "solaredge",
  "display_name": "SE30K",
  "enabled": true,
  "inverter": {"ac_power_w": 12300, "ac_voltage_v": 230.5, ...},
  "phases": {"l1": {...}, "l2": {...}, "l3": {...}},
  "connection": {"state": "connected", ...},
  ...
}
```

GET /api/devices/virtual/snapshot -> JSON:
```json
{
  "total_power_w": 13080,
  "virtual_name": "Fronius PV Inverter Proxy",
  "contributions": [
    {"device_id": "abc123", "name": "SE30K", "power_w": 12300, "throttle_order": 1, "throttle_enabled": true, "current_limit_pct": 100.0, "enabled": true},
    {"device_id": "def456", "name": "HM-800", "power_w": 780, "throttle_order": 2, "throttle_enabled": true, "current_limit_pct": 100.0, "enabled": true}
  ]
}
```

WebSocket message types:
```json
{"type": "device_snapshot", "device_id": "abc123", "data": {...}}
{"type": "virtual_snapshot", "data": {...}}
{"type": "device_list", "data": {"devices": [...]}}
```

    ```javascript
        var parts = hash.split('/');
        if (parts[0] === 'device' && parts.length >= 3) {
            return { type: 'device', id: parts[1], tab: parts[2] };

        // Legacy redirects

        if (hash === 'registers') return { type: 'device', id: _firstInverterId(), tab: 'registers' };
        return { type: 'device', id: 'virtual', tab: 'dashboard' };
    }
    ```
