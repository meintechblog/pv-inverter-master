Add method that writes to SE30K proprietary registers:

```python
async def write_power_limit(self, enable: bool, limit_pct: float) -> WriteResult:
    if self._client is None or not self._client.connected:
        return WriteResult(success=False, error="Not connected")
    try:
        # Write enable/disable to 0xF300 (62208)
        result = await self._client.write_register(
            0xF300, int(enable), slave=self.unit_id
        )
        if result.isError():
            return WriteResult(success=False, error=f"Enable write failed: {result}")

        if enable:
            # Write Float32 power limit to 0xF322-0xF323 (62242-62243)
            hi, lo = wmaxlimpct_to_se_registers(
                int(limit_pct * 100), -2  # Convert back to SunSpec raw
            )
            # Actually, write the float directly:
            packed = struct.pack(">f", limit_pct)
            hi, lo = struct.unpack(">HH", packed)
            result = await self._client.write_registers(
                0xF322, [hi, lo], slave=self.unit_id
            )
            if result.isError():
                return WriteResult(success=False, error=f"Limit write failed: {result}")

        return WriteResult(success=True)
    except Exception as e:
        return WriteResult(success=False, error=str(e))
```

```python
async def async_setValues(self, fc_as_hex, address, values):
    """Intercept writes to Model 123 registers for power control."""
    # pymodbus passes 0-based address. For holding registers with DATABLOCK_START=40001,
    # register 40154 arrives as address 40153 (40154 - 1).
    # Convert to absolute SunSpec address: address + 1 (pymodbus offset)
    abs_addr = address + 1  # Back to SunSpec addressing

    if self._control.is_model_123_address(abs_addr, len(values)):
        await self._handle_control_write(abs_addr, values)
        return

    # Default: store in datablock
    self.setValues(fc_as_hex, address, values)
```

3. Add `_handle_control_write` method:
```python
async def _handle_control_write(self, abs_addr: int, values: list[int]):
    """Process a write to Model 123 control registers."""
    from venus_os_fronius_proxy.control import (
        MODEL_123_START, WMAXLIMPCT_OFFSET, WMAXLIM_ENA_OFFSET,
        validate_wmaxlimpct,
    )

    offset = abs_addr - MODEL_123_START

    # Handle WMaxLimPct write (offset 5, register 40154)
    if offset == WMAXLIMPCT_OFFSET and len(values) >= 1:
        error = validate_wmaxlimpct(values[0])
        if error:
            raise Exception(f"ILLEGAL_VALUE: {error}")

        self._control.update_wmaxlimpct(values[0])

        if self._control.is_enabled:
            result = await self._plugin.write_power_limit(
                True, self._control.wmaxlimpct_float
            )
            if not result.success:
                raise Exception(f"Write failed: {result.error}")

        # Update local readback registers
        readback = self._control.get_model_123_readback()
        # Write full Model 123 block to datablock for read-back
        # Model 123 starts at datablock address 40150 (40149 + 1 pymodbus offset)
        self.store['h'].setValues(40150, readback)
        return

    # Handle WMaxLim_Ena write (offset 9, register 40158)
    if offset == WMAXLIM_ENA_OFFSET and len(values) >= 1:
        ena_value = values[0]
        if ena_value not in (0, 1):
            raise Exception(f"ILLEGAL_VALUE: WMaxLim_Ena must be 0 or 1, got {ena_value}")

        self._control.update_wmaxlim_ena(ena_value)

        result = await self._plugin.write_power_limit(
            self._control.is_enabled, self._control.wmaxlimpct_float
        )
        if not result.success:
            raise Exception(f"Write failed: {result.error}")

        # Update local readback
        readback = self._control.get_model_123_readback()
        self.store['h'].setValues(40150, readback)
        return

    # Other Model 123 registers: store locally only (no SE30K forwarding)
    # Use the default setValues for local storage
    self.store['h'].setValues(abs_addr - DATABLOCK_START + 1, values)
```
